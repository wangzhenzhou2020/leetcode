[131.分割回文串](https://leetcode-cn.com/problems/palindrome-partitioning/)
```
class Solution:
    def __init__(self):
        self.res = []
        self.list_str = None
        self.len = None
    def partition(self, s: str) -> List[List[str]]:
        def is_huiwen(lis: str):
            length = len(lis)
            if length == 0: return False
            for i in range(length//2):
                if lis[i]!=lis[length-i-1]:
                    return False
            return True

        def backTracking(temp,temp_str_num,start):
            if temp_str_num == self.len:
                self.res.append(temp[:])
            
            for end in range(start,self.len): # 闭区间 [start,self.len-1]
                slic = self.list_str[start:end+1]    # 截取 闭区间[start,end]            
                if is_huiwen(slic): 
                    temp.append(''.join(slic))  # append.字符串
                    temp_str_num+=len(slic)
                    backTracking(temp,temp_str_num,end+1)   # 下一段从end+1开始
                    temp_str_num-=len(slic)
                    temp.pop() # 
                else:
                    continue
               
        self.list_str = list(s)
        self.len = len(s)
        backTracking([],0,0)
        return self.res
```
