class Solution:
    def __init__(self):
        self.res = []

    def subsetsWithDup(self, nums: List[int]) -> List[List[int]]:
        def backTracking(temp,start):
            self.res.append(temp[:])

            for i in range(start,len(nums)):
                if i>start and nums[i]==nums[i-1]: continue            
                else:
                    temp.append(nums[i])
                    backTracking(temp,i+1)
                    temp.pop()

        
        nums.sort()
        backTracking([],0)
        return self.res
