[40. 组合总和 II](https://leetcode-cn.com/problems/combination-sum-ii/)
```
class Solution:
    def __init__(self):
        self.res = []

    def combinationSum2(self, candidates: List[int], target: int) -> List[List[int]]:
        def bachTracking(temp,temp_sum,used,start):
            if temp_sum == target:
                self.res.append(temp[:])
                return
            
            for i in range(start,len(candidates)):
                # e.g. used = [1,0,1]: 1 前面的0：曾经入过temp，现在不在temp            
                if i > 0 and used[i-1]==0 and candidates[i]==candidates[i-1]:continue               
                if candidates[i]+temp_sum<=target:
                    temp.append(candidates[i])
                    temp_sum+=candidates[i]
                    used[i]=1  # candidate[i] 在temp中
                    bachTracking(temp,temp_sum,used,i+1)
                    used[i]=0
                    temp.pop()
                    temp_sum-=candidates[i]
                else:
                    return

        candidates.sort() # 为了剪枝
        used = [0 for i in range(len(candidates))]
        bachTracking([],0,used,0)
        return self.res
```
