class Solution:
    def __init__(self):
        self.res = []
        
    def subsets(self, nums: List[int]) -> List[List[int]]:
        def backTracking(temp,start):
            self.res.append(temp[:])

            for i in range(start,len(nums)):
                temp.append(nums[i])
                backTracking(temp,i+1)
                temp.pop()
        
        backTracking([],0)
        
        
        return self.res
