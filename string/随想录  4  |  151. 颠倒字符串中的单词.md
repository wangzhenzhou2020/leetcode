class Solution:
    def reverseWords(self, s: str) -> str:
        # approach
        # 1 clean '',the element of list is str,not char
        # for save space.
        # 2 reverse list
        # 3 return
        l_str = []

        left,right = 0,0
        length = len(s)
        left_flag,right_flag = 0,0
        for i in range(length):
            if i==0 and s[i]!=' ':
                left = 0
                left_flag = 1
            elif s[i]!=' ' and s[i-1]==' ':
                left = i
                left_flag = 1

            elif s[i]==' ' and s[i-1]!=' ':
                right = i
                right_flag = 1
            elif i==length-1 and s[i]!=' ':
                right = length
                right_flag = 1
            
            if left_flag and right_flag:     
                l_str.append(s[left:right])
                l_str.append(' ')
                left_flag,right_flag = 0,0
        
        # 最后 的 ' '
        l_str.pop()
        l_str.reverse()
        return ''.join(l_str)
