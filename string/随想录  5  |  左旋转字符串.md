# 上述两种方式应该都申请额外空间了，假设不允许申请额外与n相关的空间：
#  随想录妙解：
class Solution:
    def reverseLeftWords(self, s: str, n: int) -> str:
        l_s = list(s)
        l_s.reverse()
        
        # l_s[0:len(s)-n].reverse() # 这样应该是创建了新的list了
        left = 0
        right = len(s)-n-1
        while(left<right):
            l_s[left],l_s[right]=l_s[right],l_s[left]
            left += 1
            right -= 1
        left = len(s)-n
        right = len(s)-1
        while(left<right):
            l_s[left],l_s[right]=l_s[right],l_s[left]
            left += 1
            right -= 1

        return ''.join(l_s)
