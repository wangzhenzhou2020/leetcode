```
class Solution(object):
    def minSubArrayLen(self, target, nums):
        """
        :type target: int
        :type nums: List[int]
        :rtype: int
        """
        # 暴力 o(n^2)

        # shortest_len = 0
        # for i in range(len(nums)):
        #     sum = 0
        #     for j in range(i,len(nums)):
                
        #         sum += nums[j]
        #         if target <= sum :  # 有长度
        #             shortest_len_i =  j-i+1
        #             if shortest_len == 0: # 第一次
        #                 shortest_len = shortest_len_i
        #             else:
        #                 shortest_len = min(shortest_len_i,shortest_len)
        #             break
           
        # return shortest_len

        # 滑动窗口：避免无意义地移动两个指针
        if nums==None or len(nums)==0: return 0
        if len(nums)==1:
            return  0 if nums[0]<target else 1

        res = 1e6  # 1 res = float("inf")
        slow , fast = 0,0 # 闭区间
        sum = nums[slow]
        
        while(slow<=len(nums)-1):# 每次循环，依据sum动一个游标，动一次。
            
            if target<=sum : # slow 前进
                a = fast-slow + 1
                res = min(res,a)                
                sum -= nums[slow] 
                slow += 1 
            else:            # fast 前进  
                fast += 1
                if fast <= len(nums)-1:
                    sum += nums[fast]
                else:         # sum 不足，且fast已经到头，
                    break
        
        return 0 if res==1e6 else res
```

## 滑动窗口，属于双指针方法，避免不必要的指针移动，即不必要的前进和回退。
